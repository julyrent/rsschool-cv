void selectionSort(int arr[], int size)
{
	int min, sum(0);
	for (size_t i(0); i < size; ++i)
	{
		++sum;
		min = i;
		for (size_t j(i + 1); j < size; ++j)
		{
			if (arr[j] < arr[min])
			{
				min = j;
			}
			++sum;
		}
		int temp = arr[i];
		arr[i] = arr[min];
		arr[min] = temp;
	}
	cout << "Selection Time: " << sum << endl;
}
